00300         var sign = (a * d < 0.0 || b * c > 0.0) ? -1.0 : 1.0, \
00301             a2 = (ATAN2(b, d) + ATAN2(-sign * c, sign * a)) / 2.0, \
00302             cos = COS(a2),\
00303             sin = SIN(a2);\
00304         \
00305         if (cos == 0)\
00306         {\
00307             sx = -c / sin;\
00308             sy = b / sin;\
00309         }\
00310         else if (sin == 0)\
00311         {\
00312             sx = a / cos;\
00313             sy = d / cos;\
00314         }\

00318             abs_sin = ABS(sin);\

00299     var bounds = [self bounds];
00300 
00301     switch (_boxType)

00304             // NSBox does not include a horizontal flag for the separator type. We have to determine

00306             if (CGRectGetWidth(bounds) === 5.0)
00307                 return [self _drawVerticalSeperatorInRect:bounds];
00308             else if (CGRectGetHeight(bounds) === 5.0)
00309                 return [self _drawHorizontalSeperatorInRect:bounds];
00310 
00311             break;
00312     }
00313

00316         case CPBezelBorder:
00317             [self _drawBezelBorderInRect:bounds];

00319
